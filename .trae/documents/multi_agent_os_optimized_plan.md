## 配置模板

### Discord 通道配置
```json
{
  "channels": {
    "discord": {
      "enabled": true,
      "botToken": "${DISCORD_BOT_TOKEN}",
      "guildId": "${DISCORD_GUILD_ID}",
      "channels": {
        "运营频道": "1234567890",
        "技术频道": "0987654321"
      }
    }
  }
}
```

### 智能体间通信配置
```json
{
  "agentCommunication": {
    "enabled": true,
    "permissions": {
      "main": {
        "canCommand": ["coo", "business", "life"],
        "canForwardTo": ["feishu:核心管理群"]
      },
      "coo": {
        "canCommand": ["business"],
        "canForwardTo": ["discord:运营频道"]
      }
    }
  }
}
```

### 路由绑定配置
```json
{
  "bindings": [
    {
      "match": {
      "channel": "feishu",
      "chatType": "group",
      "groupId": "核心管理群"
    },
    "route": {
      "agent": "main"
    }
  },
  {
    "match": {
      "channel": "discord",
      "channelId": "1234567890"
    },
    "route": {
      "agent": "coo"
    }
  },
  {
    "match": {
      "channel": "discord",
      "channelId": "0987654321"
    },
    "route": {
      "agent": "business"
    }
  }
}
```
